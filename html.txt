  function loadMonthDropdown(){
    google.script.run
      .withSuccessHandler(function(months){
        const select = document.getElementById("monthSelect");
        select.innerHTML = "";

        if(!months || !months.length){
          select.innerHTML = `<option value="MEI 26">MEI 26</option>`;
          return;
        }

        months.forEach(name => {
          const opt = document.createElement("option");
          opt.value = name;
          opt.textContent = name;
          select.appendChild(opt);
        });

        if(months.includes("MEI 26")){
          select.value = "MEI 26";
        }

        currentSheetName = select.value;
      })
      .getAvailableMonths();
  }

  function loadPayroll(){
    currentPage = "payroll";

    const sheetName = document.getElementById("monthSelect").value || "MEI 26";
    currentSheetName = sheetName;

    document.getElementById("activeSheetName").textContent = sheetName;
    document.getElementById("tableContainer").innerHTML = "Loading payroll...";
    setMsg("");

    google.script.run
      .withSuccessHandler(function(res){
        if(!res || !res.ok){
          document.getElementById("tableContainer").innerHTML = "";
          setMsg(res ? res.message : "Failed to load payroll.");
          return;
        }

        renderPayrollTable(res);
      })
      .withFailureHandler(function(err){
        setMsg(err.message || "Failed to load payroll.");
      })
      .getPayrollData(sheetName);
  }

  function renderPayrollTable(res){
  lastPayrollResponse = res;

  const data = res.data || [];
  const bg = res.backgrounds || [];
  const fc = res.fontColors || [];

  if(!data.length){
    document.getElementById("tableContainer").innerHTML = "No data.";
    return;
  }

  currentData = data;
  currentHeaders = data[0];
  payrollRowsCache = [];

  const normalizedHeaders = currentHeaders.map(h => normalizeHeader(h));

  const payslipIndex = normalizedHeaders.indexOf("payslip type");
  const staffIdIndex = normalizedHeaders.indexOf("no. pekerja");
  const namaIndex = normalizedHeaders.indexOf("nama");

  let html = `
    <table id="dataTable">
      <thead>
        <tr>
  `;

  currentHeaders.forEach((h, i) => {
    html += `
      <th style="
        background:${bg[0][i]};
        color:${fc[0][i]};
        font-weight:900;
      ">
        ${escapeHtml(h)}
      </th>
    `;
  });

  html += `
        </tr>
      </thead>
      <tbody>
  `;

  for(let r = 1; r < data.length; r++){
    const row = data[r];

    const staffId = String(row[staffIdIndex] || "").trim();
    const nama = String(row[namaIndex] || "").trim();

    if(!staffId || !nama) continue;

    const realSheetRow = res.headerRow + r;

    payrollRowsCache.push({
      rowNumber: realSheetRow,
      values: row
    });

    html += `<tr data-row-number="${realSheetRow}">`;

    row.forEach((cell, c) => {
      const headerName = normalizeHeader(currentHeaders[c]);

      if(c === payslipIndex && isEditMode){
        html += `
          <td style="background:${bg[r][c]}; color:${fc[r][c]};">
            <select class="templateDropdown payrollInput" data-col="${c}">
              <option value="">PaySlip Type</option>

              ${templateOptions.map(t => `
                <option value="${escapeAttr(t)}"
                  ${cell === t ? "selected" : ""}>
                  ${escapeHtml(t)}
                </option>
              `).join("")}

            </select>
          </td>
        `;
      }

      else if(c === payslipIndex){
        html += `
          <td style="
            background:${bg[r][c]};
            color:${fc[r][c]};
            font-weight:700;
          ">
            ${escapeHtml(cell || "-")}
          </td>
        `;
      }

      else if(isEditMode){
        const lockedColumns = [
          "jumlah ot",
          "total"
        ];

        const isLocked =
          lockedColumns.includes(headerName) &&
          c <= 12;

        if(isLocked){
          html += `
            <td style="
              background:${bg[r][c]};
              color:${fc[r][c]};
              font-weight:700;
              opacity:.7;
            ">
              ${escapeHtml(cell)}
            </td>
          `;
        }

        else{
          html += `
            <td style="background:${bg[r][c]}; color:${fc[r][c]};">
              <input
                class="payrollInput"
                data-col="${c}"
                value="${escapeAttr(cell)}">
            </td>
          `;
        }
      }

      else {
        if(headerName === "file stored link" && cell){
          html += `
            <td style="
              background:${bg[r][c]};
              color:${fc[r][c]};
            ">
              <a
                href="${escapeAttr(cell)}"
                target="_blank"
                style="
                  color:#2563eb;
                  font-weight:700;
                  text-decoration:none;
                ">
                Open File
              </a>
            </td>
          `;
        }

        else{
          html += `
            <td style="
              background:${bg[r][c]};
              color:${fc[r][c]};
            ">
              ${escapeHtml(cell)}
            </td>
          `;
        }
      }
    });

    html += `</tr>`;
  }

  html += `
      </tbody>
    </table>
  `;

  document.getElementById("tableContainer").innerHTML = html;

  updateStats(data);
}

  function renderNormalTable(res){
    const data = res.data || [];
    const bg = res.backgrounds || [];
    const fc = res.fontColors || [];

    if(!data.length){
      document.getElementById("tableContainer").innerHTML = "No data.";
      return;
    }

    let html = `<table id="dataTable"><thead><tr>`;

    data[0].forEach((h, i) => {
      html += `
        <th style="background:${bg[0][i]}; color:${fc[0][i]};">
          ${escapeHtml(h)}
        </th>
      `;
    });

    html += `</tr></thead><tbody>`;

    for(let r = 1; r < data.length; r++){
      const row = data[r];

      if(row.join("").trim() === "") continue;

      html += `<tr>`;

      row.forEach((cell, c) => {
        html += `
          <td style="background:${bg[r][c]}; color:${fc[r][c]};">
            ${escapeHtml(cell)}
          </td>
        `;
      });

      html += `</tr>`;
    }

    html += `</tbody></table>`;

    document.getElementById("tableContainer").innerHTML = html;
  }

  function updateStats(data){
  if(!data || data.length < 2) return;

  const headers = data[0].map(h => normalizeHeader(h));

  const staffIdIndex = headers.indexOf("no. pekerja");
  const namaIndex = headers.indexOf("nama");
  const basicIndex = headers.indexOf("basic");
  const netIndex = headers.indexOf("net payment");

  const kwspIndex = headers.indexOf("kwsp 11%");
  const perkesoIndex = headers.indexOf("socso");
  const sipIndex = headers.indexOf("sip");

  let staff = 0;
  let totalBasic = 0;
  let totalNet = 0;
  let totalDeduct = 0;

  for(let i = 1; i < data.length; i++){
    const row = data[i];

    const staffId = String(row[staffIdIndex] || "").trim();
    const nama = String(row[namaIndex] || "").trim();

    if(!staffId || !nama) continue;

    staff++;

    totalBasic += toNumber(row[basicIndex]);
    totalNet += toNumber(row[netIndex]);

    totalDeduct +=
      toNumber(row[kwspIndex]) +
      toNumber(row[perkesoIndex]) +
      toNumber(row[sipIndex]);
  }

  document.getElementById("statStaff").textContent = staff;
  document.getElementById("statBasic").textContent = formatMoney(totalBasic);
  document.getElementById("statNet").textContent = formatMoney(totalNet);
  document.getElementById("statDeduct").textContent = formatMoney(totalDeduct);
}

function normalizeHeader(value){
  return String(value || "")
    .replace(/\n/g, " ")
    .replace(/\s+/g, " ")
    .trim()
    .toLowerCase();
}

  function toNumber(value){
    if(value === undefined || value === null) return 0;

    const cleaned = String(value)
      .replace(/RM/gi, "")
      .replace(/,/g, "")
      .replace(/-/g, "0")
      .trim();

    const n = Number(cleaned);
    return isNaN(n) ? 0 : n;
  }

  function formatMoney(num){
    return Number(num || 0).toLocaleString("en-MY", {
      minimumFractionDigits: 2,
      maximumFractionDigits: 2
    });
  }

  function escapeHtml(value){
    return String(value)
      .replace(/&/g,"&amp;")
      .replace(/</g,"&lt;")
      .replace(/>/g,"&gt;")
      .replace(/"/g,"&quot;")
      .replace(/'/g,"&#039;");
  }

function updatePayrollTable(){
  const rows = [];

  document.querySelectorAll("#dataTable tbody tr").forEach(tr => {
    const rowNumber = Number(tr.dataset.rowNumber);
    const oldRow = payrollRowsCache.find(r => r.rowNumber === rowNumber);

    if(!oldRow) return;

    const values = [...oldRow.values];

    tr.querySelectorAll(".payrollInput").forEach(input => {
      const col = Number(input.dataset.col);
      values[col] = input.value;
    });

    rows.push({
      rowNumber,
      values
    });
  });

  setMsg("Updating table...");

  google.script.run
    .withSuccessHandler(function(res){
      setMsg(res && res.ok ? "Table updated." : "Failed to update table.");

      isEditMode = false;
      document.getElementById("cancelEditBtn").classList.add("hidden");
      document.getElementById("editTableBtn").textContent = "Edit Table";

      loadPayroll();
    })
    .withFailureHandler(function(err){
      setMsg(err.message || "Failed to update table.");
    })
    .updatePayrollRows(currentSheetName, rows);
}
